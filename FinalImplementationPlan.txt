  Schema::table('disputes', function (Blueprint $table) {
      // Polymorphic target (replaces company_id as primary target)
      $table->string('disputable_type')->nullable()->after('id');
      $table->unsignedBigInteger('disputable_id')->nullable()->after('disputable_type');
      $table->index(['disputable_type', 'disputable_id']);

      // Risk-ordered type (A/B/C/D)
      $table->string('type', 1)->default('A')->after('disputable_id');
      $table->index('type');

      // Snapshot linkage
      $table->unsignedBigInteger('snapshot_id')->nullable()->after('type');
      $table->foreign('snapshot_id')->references('id')->on('dispute_snapshots');

      // Integrity tracking (computed, not stored blindly)
      $table->string('integrity_status')->default('pending')->after('snapshot_id');

      // Disputed amount (frozen at filing)

      // Settlement tracking
      $table->string('settlement_type')->nullable();
      $table->unsignedBigInteger('settlement_reference_id')->nullable();
  });

  3. ENUM DEFINITIONS

  DisputeStatus.php

  <?php

  namespace App\Enums;

  enum DisputeStatus: string
  {
      case OPEN = 'open';
      case UNDER_REVIEW = 'under_review';
      case AWAITING_INVESTOR = 'awaiting_investor';
      case ESCALATED = 'escalated';
      case RESOLVED_APPROVED = 'resolved_approved';
      case RESOLVED_REJECTED = 'resolved_rejected';
      case CLOSED = 'closed';

      public function isTerminal(): bool
      {
          return $this === self::CLOSED;
      }

      public function isResolved(): bool
      {
          return in_array($this, [self::RESOLVED_APPROVED, self::RESOLVED_REJECTED]);
      }

      public function allowsRefund(): bool
      {
          return $this === self::RESOLVED_APPROVED;
      }

      public static function activeStates(): array
      {
          return [self::OPEN, self::UNDER_REVIEW, self::AWAITING_INVESTOR, self::ESCALATED];
      }
  }

  DisputeType.php

  <?php

  namespace App\Enums;

  use App\Models\Payment;
  use App\Models\UserInvestment;
  use App\Models\Withdrawal;
  use App\Models\BonusTransaction;

  enum DisputeType: string
  {
      case TYPE_A_CONFUSION = 'A';      // Lowest risk - user confusion
      case TYPE_B_PAYMENT = 'B';        // Payment failure inconsistency
      case TYPE_C_ALLOCATION = 'C';     // Allocation mismatch
      case TYPE_D_FRAUD = 'D';          // Fraud/Chargeback - Highest risk

      public function riskLevel(): int
      {
          return match($this) {
              self::TYPE_A_CONFUSION => 1,
              self::TYPE_B_PAYMENT => 2,
              self::TYPE_C_ALLOCATION => 3,
              self::TYPE_D_FRAUD => 4,
          };
      }

      public function requiresAutoEscalation(): bool
      {
          return $this === self::TYPE_D_FRAUD;
      }

      public function allowedDisputableTypes(): array
      {
          return match($this) {
              self::TYPE_A_CONFUSION => [UserInvestment::class, BonusTransaction::class],
              self::TYPE_B_PAYMENT => [Payment::class],
              self::TYPE_C_ALLOCATION => [UserInvestment::class],
              self::TYPE_D_FRAUD => [Payment::class, Withdrawal::class],
          };
      }

      public function isValidDisputable(string $disputableType): bool
      {
          return in_array($disputableType, $this->allowedDisputableTypes());
      }

      public function label(): string
      {
          return match($this) {
              self::TYPE_A_CONFUSION => 'User Confusion',
              self::TYPE_B_PAYMENT => 'Payment Failure',
              self::TYPE_C_ALLOCATION => 'Allocation Mismatch',
              self::TYPE_D_FRAUD => 'Fraud / Chargeback',
          };
      }
  }

  class DisputeStateMachine
  {
      // Transition matrix - defines allowed state transitions
      private const TRANSITIONS = [
          'open' => ['under_review', 'escalated'],
          'under_review' => ['awaiting_investor', 'escalated', 'resolved_approved', 'resolved_rejected'],
          'awaiting_investor' => ['under_review', 'escalated'],
          'escalated' => ['resolved_approved', 'resolved_rejected'],
          'resolved_approved' => ['closed'],
          'resolved_rejected' => ['closed'],
          'closed' => [], // Terminal - no transitions
      ];

      public function canTransition(DisputeStatus $from, DisputeStatus $to): bool;
      public function transition(Dispute $dispute, DisputeStatus $to, int $actorId, string $reason): Dispute;
      public function validateTransition(DisputeStatus $from, DisputeStatus $to): void; // Throws on invalid
      public function getAvailableTransitions(DisputeStatus $current): array;
      public function isDowngrade(DisputeStatus $from, DisputeStatus $to): bool;
  }

  5. TRANSITION MATRIX ENFORCEMENT APPROACH

  ┌─────────────────────┬────────────────────────────────────────────────────────┐
  │ Current State       │ Allowed Transitions                                    │
  ├─────────────────────┼────────────────────────────────────────────────────────┤
  │ OPEN                │ → UNDER_REVIEW, ESCALATED                              │
  │ UNDER_REVIEW        │ → AWAITING_INVESTOR, ESCALATED, RESOLVED_*             │
  │ AWAITING_INVESTOR   │ → UNDER_REVIEW, ESCALATED                              │
  │ ESCALATED           │ → RESOLVED_APPROVED, RESOLVED_REJECTED (NO DOWNGRADE)  │
  │ RESOLVED_APPROVED   │ → CLOSED                                               │
  │ RESOLVED_REJECTED   │ → CLOSED                                               │
  │ CLOSED              │ (TERMINAL - NO TRANSITIONS)                            │
  └─────────────────────┴────────────────────────────────────────────────────────┘

  ENFORCEMENT LAYERS:
  1. DisputeStateMachine::validateTransition() - throws InvalidStateTransitionException
  2. Dispute model boot() - prevents direct status changes, forces use of state machine
  3. Controller layer - validates before attempting transition
  4. Database trigger (optional) - last line of defense

  AUTO-ESCALATION TRIGGERS:
  - DisputeType::TYPE_D_FRAUD → always escalate on creation
  - SnapshotIntegrityService returns 'compromised' → auto-escalate
  - disputed_amount_paise > config('dispute.escalation_threshold') → auto-escalate
  - Ledger imbalance detected → auto-escalate

  DOWNGRADE PREVENTION:
  - ESCALATED cannot transition to OPEN, UNDER_REVIEW, or AWAITING_INVESTOR
  - DisputeStateMachine::isDowngrade() returns true for these attempts
  - Blocked at validation layer with clear exception message

  Computed by:
  class DisputePermissionResolver
  {
      public function resolve(Dispute $dispute, User $admin): array
      {
          $status = $dispute->status;
          $stateMachine = app(DisputeStateMachine::class);

          return [
              'can_transition' => !$status->isTerminal(),
              'can_escalate' => $stateMachine->canTransition($status, DisputeStatus::ESCALATED),
              'can_resolve' => in_array($status, [DisputeStatus::UNDER_REVIEW, DisputeStatus::ESCALATED]),
              'can_override_defensibility' => !$status->isTerminal() && !$status->isResolved(),
              'can_refund' => $status === DisputeStatus::RESOLVED_APPROVED && !$dispute->isSettled(),
              'can_close' => $status->isResolved(),
              'available_transitions' => $stateMachine->getAvailableTransitions($status),
          ];
      }
  }
